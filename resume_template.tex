\newcommand{\ressection}[4]{
    \vspace{12pt}
	% #1 Is the job's title and #2 the company's name'

	% #3 is the job's geographical location and #4 is the date
	\textit{{\footnotesize\bfseries #3\hfill\MakeUppercase{#4}}}\par
	\vspace{6pt}
}
